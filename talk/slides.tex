\tikzmath
{
		  function symlog(\x,\a){
		    \yLarge = ((\x>\a) - (\x<-\a)) * (ln(max(abs(\x/\a),1)) + 1);
		    \ySmall = (\x >= -\a) * (\x <= \a) * \x / \a ;
		    return \yLarge + \ySmall ;
		  };
		  function symexp(\y,\a){
		    \xLarge = ((\y>1) - (\y<-1)) * \a * exp(abs(\y) - 1) ;
		    \xSmall = (\y>=-1) * (\y<=1) * \a * \y ;
		    return \xLarge + \xSmall ;
		  };
}
